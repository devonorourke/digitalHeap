#create tab-delimited file from .fasta:
from Bio import SeqIO
count = SeqIO.convert("scaffolds.fasta", "fasta", "scaffolds.txt", "tab")
print("Converted %i records" % count)
#then select just the first sequence:
head -n 1 scaffolds.txt | awk '{print $2}'
#then count scaf lengths:
awk '{print length($2)}' scaffolds.txt > scaflengths.txt
paste scaffolds.txt scaflengths.txt > scafData.txt
